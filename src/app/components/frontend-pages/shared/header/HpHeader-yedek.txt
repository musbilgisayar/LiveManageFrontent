"use client";
import React, { useState } from "react";
import AppBar from "@mui/material/AppBar";
import Button from "@mui/material/Button";
import Container from "@mui/material/Container";
import Drawer from "@mui/material/Drawer";
import IconButton from "@mui/material/IconButton";
import Stack from "@mui/material/Stack";
import Toolbar from "@mui/material/Toolbar";
import useMediaQuery from "@mui/material/useMediaQuery";
import { styled } from "@mui/material/styles";
import Logo from "@/app/[locale]/(DashboardLayout)/layout/shared/logo/Logo";
import Navigations from "./Navigations";
import MobileSidebar from "./MobileSidebar";
import { IconMenu2 } from "@tabler/icons-react";
import { Box } from "@mui/material";

// 🔑 Yeni tablı form
import AuthTabs from "@/app/[locale]/(frontend-pages)/auth/authForms/AuthTabs";

const HpHeader = () => {
  const AppBarStyled = styled(AppBar)(({ theme }) => ({
    justifyContent: "center",
    [theme.breakpoints.up("lg")]: {
      minHeight: "100px",
    },
    backgroundColor: theme.palette.primary.light,
  }));

  const ToolbarStyled = styled(Toolbar)(({ theme }) => ({
    width: "100%",
    paddingLeft: "0 !important",
    paddingRight: "0 !important",
    color: theme.palette.text.secondary,
    justifyContent: "space-between",
  }));

  // sidebar states
  const lgUp = useMediaQuery((theme: any) => theme.breakpoints.up("lg"));
  const lgDown = useMediaQuery((theme: any) => theme.breakpoints.down("lg"));

  const [menuOpen, setMenuOpen] = useState(false);   // sol menü
  const [loginOpen, setLoginOpen] = useState(false); // sağ login drawer

  return (
    <AppBarStyled position="sticky" elevation={0}>
      <Container sx={{ maxWidth: "1400px !important" }}>
        <ToolbarStyled>
          <Logo />
          {lgDown ? (
            <IconButton
              color="inherit"
              aria-label="menu"
              onClick={() => setMenuOpen(true)}
            >
              <IconMenu2 size="20" />
            </IconButton>
          ) : null}

          {lgUp ? (
            <>
              <Stack spacing={1} direction="row" alignItems="center">
                <Navigations />
              </Stack>
              <Button
                color="primary"
                variant="contained"
                onClick={() => setLoginOpen(true)}
              >
                Log In
              </Button>
            </>
          ) : null}
        </ToolbarStyled>
      </Container>

      {/* Sol menü (mobil sidebar) */}
      <Drawer
        anchor="left"
        open={menuOpen}
        variant="temporary"
        onClose={() => setMenuOpen(false)}
        slotProps={{
          paper: {
            sx: {
              width: 270,
              border: "0 !important",
              boxShadow: (theme) => theme.shadows[8],
            },
          },
        }}
      >
        <MobileSidebar />
      </Drawer>

      {/* Sağdan kayan Auth Drawer */}
      <Drawer
        anchor="right"
        open={loginOpen}
        onClose={() => setLoginOpen(false)}
      >
        <Box sx={{ width: 400, p: 3 }}>
          {/* 👇 Tek component: Login + Register tablı form */}
          <AuthTabs />

          <Button
            onClick={() => setLoginOpen(false)}
            sx={{ mt: 2 }}
            variant="outlined"
            fullWidth
          >
            Kapat
          </Button>
        </Box>
      </Drawer>
    </AppBarStyled>
  );
};

export default HpHeader;

import { IconMenu2 } from "@tabler/icons-react";
import { Box, Typography } from "@mui/material";

// 🔑 auth formları
import AuthLogin from "@/app/[locale]/(frontend-pages)/auth/authForms/AuthLogin";
import AuthRegister from "@/app/[locale]/(frontend-pages)/auth/authForms/AuthRegister";
import AuthForgotPassword from "@/app/[locale]/(frontend-pages)/auth/authForms/AuthForgotPassword";
import AuthTwoSteps from "@/app/[locale]/(frontend-pages)/auth/authForms/AuthTwoSteps";

const HpHeader = () => {
  const AppBarStyled = styled(AppBar)(({ theme }) => ({
    justifyContent: "center",
    [theme.breakpoints.up("lg")]: {
      minHeight: "100px",
    },
    backgroundColor: theme.palette.primary.light,
  }));

  const ToolbarStyled = styled(Toolbar)(({ theme }) => ({
    width: "100%",
    paddingLeft: "0 !important",
    paddingRight: "0 !important",
    color: theme.palette.text.secondary,
    justifyContent: "space-between",
  }));

  // sidebar states
  const lgUp = useMediaQuery((theme: any) => theme.breakpoints.up("lg"));
  const lgDown = useMediaQuery((theme: any) => theme.breakpoints.down("lg"));

  const [menuOpen, setMenuOpen] = useState(false);   // sol menü
  const [loginOpen, setLoginOpen] = useState(false); // sağ login drawer
  const [authView, setAuthView] = useState<"login" | "register" | "forgot" | "twoSteps">("login");

  return (
    <AppBarStyled position="sticky" elevation={0}>
      <Container sx={{ maxWidth: "1400px !important" }}>
        <ToolbarStyled>
          <Logo />
          {lgDown ? (
            <IconButton
              color="inherit"
              aria-label="menu"
              onClick={() => setMenuOpen(true)}
            >
              <IconMenu2 size="20" />
            </IconButton>
          ) : null}

          {lgUp ? (
            <>
              <Stack spacing={1} direction="row" alignItems="center">
                <Navigations />
              </Stack>
              <Button
                color="primary"
                variant="contained"
                onClick={() => {
                  setAuthView("login");   // 🔑 Drawer her açıldığında login’den başla
                  setLoginOpen(true);
                }}
              >
                Log In
              </Button>
            </>
          ) : null}
        </ToolbarStyled>
      </Container>

      {/* Sol menü (mobil sidebar) */}
      <Drawer
        anchor="left"
        open={menuOpen}
        variant="temporary"
        onClose={() => setMenuOpen(false)}
        slotProps={{
          paper: {
            sx: {
              width: 270,
              border: "0 !important",
              boxShadow: (theme) => theme.shadows[8],
            },
          },
        }}
      >
        <MobileSidebar />
      </Drawer>

      {/* Sağdan kayan Auth Drawer */}
      <Drawer
        anchor="right"
        open={loginOpen}
        onClose={() => setLoginOpen(false)}
      >
        <Box sx={{ width: 400, p: 3 }}>
          {authView === "login" && <AuthLogin onChangeView={setAuthView} />}

          {authView === "register" && (
            <AuthRegister
              onChangeView={(view) => {
                if (view === "close") {
                  setLoginOpen(false); // ✅ Drawer kapat
                } else {
                  setAuthView(view);   // login/register/forgot/twoSteps
                }
              }}
            />
          )}

          {authView === "forgot" && <AuthForgotPassword onChangeView={setAuthView} />}
          {authView === "twoSteps" && <AuthTwoSteps onChangeView={setAuthView} />}

          <Button
            onClick={() => setLoginOpen(false)}
            sx={{ mt: 2 }}
            variant="outlined"
            fullWidth
          >
            Kapat
          </Button>
        </Box>
      </Drawer>
    </AppBarStyled>
  );
};

export default HpHeader;

import React from "react";

import { styled } from "@mui/material/styles";
import Logo from "@/app/[locale]/(DashboardLayout)/layout/shared/logo/Logo"
import Navigations from "./Navigations";
import MobileSidebar from "./MobileSidebar";
import { IconMenu2 } from "@tabler/icons-react";

const HpHeader = (props: any) => {
  const AppBarStyled = styled(AppBar)(({ theme }) => ({
    justifyContent: "center",
    [theme.breakpoints.up("lg")]: {
      minHeight: "100px",
    },
    backgroundColor: theme.palette.primary.light,
  }));

  const ToolbarStyled = styled(Toolbar)(({ theme }) => ({
    width: "100%",
    paddingLeft: "0 !important",
    paddingRight: "0 !important",
    color: theme.palette.text.secondary,
    justifyContent: "space-between",
  }));

  //   sidebar
  const lgUp = useMediaQuery((theme) => theme.breakpoints.up("lg"));
  const lgDown = useMediaQuery((theme) => theme.breakpoints.down("lg"));

  const [open, setOpen] = React.useState(false);

  const handleDrawerOpen = () => {
    setOpen(true);
  };

  const toggleDrawer = (newOpen: any) => () => {
    setOpen(newOpen);
  };

  return (
    <AppBarStyled position="sticky" elevation={0}>
      <Container
        sx={{
          maxWidth: "1400px !important",
        }}
      >

            <IconButton
              color="inherit"
              aria-label="menu"
              onClick={handleDrawerOpen}
            >
              <IconMenu2 size="20" />
            </IconButton>
          ) : null}
          {lgUp ? (
            <>
              <Stack spacing={1} direction="row" alignItems="center">
                <Navigations />
              </Stack>
              <Button
                color="primary"
                variant="contained"
                href="/auth/login"
              >
                Log In
              </Button>
            </>
          ) : null}
        </ToolbarStyled>
      </Container>
      <Drawer
        anchor="left"
        open={open}
        variant="temporary"
        onClose={toggleDrawer(false)}
        slotProps={{
          paper: {
            sx: {
              width: 270,
              border: "0 !important",
              boxShadow: (theme) => theme.shadows[8],
            },
          }
        }}
      >
        <MobileSidebar />
      </Drawer>
    </AppBarStyled>
  );
};
