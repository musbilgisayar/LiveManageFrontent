"use client";

/* ===========================================================================
 * Localization Key Detail – Tüm diller için değerleri gör / düzenle
 * - /[locale]/apps/localization/[encodedKey]?ns=...
 * - Üstte mevcut key detayları (namespace + key)
 * - Aşağıda tüm diller alt alta + textbox (value)
 * - "Kaydet (Tüm Diller)" → her dil için upsert (expectedRowVersion ile)
 * - "Yeni Anahtar Ekle" → Dialog: namespace, key, tüm diller için value
 * ======================================================================== */

import React, { useCallback, useEffect, useMemo, useRef, useState } from "react";
import {
  Box,
  Button,
  Card,
  CardContent,
  Snackbar,
  Alert,
  Stack,
  TextField,
  Toolbar,
  Typography,
  Divider,
  CircularProgress,
  Dialog,
  DialogTitle,
  DialogContent,
  DialogActions,
} from "@mui/material";
import Breadcrumb from "@/app/[locale]/(DashboardLayout)/layout/shared/breadcrumb/Breadcrumb";
import { useParams, useSearchParams, useRouter } from "next/navigation";
import { API_BASE as CONFIG_API_BASE } from "@/lib/config";
import { useI18nNs } from "@/app/context/i18nContext";

const API_VERSION = "1.0";
const API_BASE = CONFIG_API_BASE || "";
const ENABLE_NET_LOG = true;

interface LanguageItem {
  cultureCode: string;
  name: string;
  flagEmoji: string;
  isDefault: boolean;
}

interface TranslationDto {
  id: number;
  key: string;
  value: string | null;
  version: string | null; // base64 rowversion
  cultureCode?: string;
}

type ValueMap = Record<string, string>; // culture -> value
type VersionMap = Record<string, string | null>;

function createCorrelationId() {
  try {
    return (crypto as any)?.randomUUID?.() ?? Math.random().toString(36).slice(2);
  } catch {
    return Math.random().toString(36).slice(2);
  }
}

function commonHeaders(extra?: HeadersInit): HeadersInit {
  return {
    accept: "application/json",
    "Content-Type": "application/json",
    "x-correlation-id": createCorrelationId(),
    "x-client-version": "lm-frontend/locmgr-1.3.0",
    "x-audit-action": "LocalizationDetail",
    "x-audit-entity": "LocalizationKey",
    "x-audit-sensitive": "none",
    ...(extra || {}),
  };
}

async function fetchJson<T>(input: RequestInfo, init?: RequestInit): Promise<T> {
  const startedAt = performance.now();
  const res = await fetch(input, {
    ...init,
    headers: commonHeaders(init?.headers || {}),
    cache: "no-store",
  });

  let payload: any = null;
  try {
    payload = await res.json();
  } catch {
    if (res.status === 204) return {} as T;
    const err = new Error(`HTTP ${res.status} (non-JSON)`) as Error & { response?: Response };
    err.response = res;
    throw err;
  } finally {
    const ms = (performance.now() - startedAt).toFixed(0);
    ENABLE_NET_LOG && console.debug("[net] response", { url: String(input), status: res.status, ms });
  }

  if (payload && typeof payload === "object" && "ok" in payload) {
    if ((payload as any).ok) return (payload as any).data as T;
    const err = new Error("API_ERROR") as Error & { response?: Response; body?: any };
    err.response = res;
    err.body = (payload as any).error ?? payload;
    throw err;
  }
  if (Array.isArray(payload)) return payload as T;
  if ((payload as any)?.data && Array.isArray((payload as any).data)) return (payload as any).data as T;

  if (!res.ok) {
    const err = new Error(`HTTP ${res.status}`) as Error & { response?: Response; body?: any };
    err.response = res;
    err.body = payload;
    throw err;
  }
  return payload as T;
}

function prettyApiError(e: any, t?: (key: string) => string): string {
  const status = e?.body?.status ?? e?.response?.status;
  const upstreamMsg =
    e?.body?.data?.message ??
    e?.body?.data?.title ??
    e?.body?.data?.detail ??
    e?.body?.message ??
    e?.message;

  const validationErrors = e?.body?.data?.errors || e?.body?.errors;
  if (validationErrors && typeof validationErrors === "object") {
    const parts = Object.entries(validationErrors).flatMap(([k, v]) =>
      Array.isArray(v) ? v.map((m) => `${k}: ${m}`) : [`${k}: ${String(v)}`]
    );
    if (parts.length) {
      return `${t?.("detailPage.errors.validation") || "Doğrulama hatası:"}\n• ${parts.join("\n• ")}`;
    }
  }

  if (e?.body?.message === "UPSTREAM_NETWORK_ERROR") {
    return t?.("detailPage.errors.upstreamNetwork") || "Sunucuya bağlanırken ağ/TLS hatası oluştu. Backend’in ayakta olduğundan emin olun.";
  }

  switch (status) {
    case 400: return `${t?.("detailPage.errors.badRequest") || "Geçersiz istek."} ${upstreamMsg ?? ""}`.trim();
    case 401: return t?.("detailPage.errors.unauthorized") || "Yetkisiz istek. Lütfen oturum açın.";
    case 403: return t?.("detailPage.errors.forbidden") || "Bu işlemi yapma yetkiniz yok (403).";
    case 404: return t?.("detailPage.errors.notFound") || "Kaynak bulunamadı (404).";
    case 409: return t?.("detailPage.errors.conflict") || "Eşzamanlı güncelleme/çakışma (409).";
    case 422: return `${t?.("detailPage.errors.unprocessable") || "İş kuralı hatası (422)."} ${upstreamMsg ?? ""}`.trim();
    case 500: return t?.("detailPage.errors.server") || "Sunucu hatası (500).";
    default: return upstreamMsg ? String(upstreamMsg) : (t?.("detailPage.errors.unexpected") || "Beklenmeyen bir hata oluştu.");
  }
}

export default function LocalizationKeyDetailPage() {
  const { t } = useI18nNs("localization");
  const router = useRouter();
  const routeParams = useParams<{ locale: string; encodedKey: string }>();
  const searchParams = useSearchParams();

  const locale = (routeParams?.locale as string) || "tr";
  const encodedKey = (routeParams?.encodedKey as string) || "";
  const fullKey = decodeURIComponent(encodedKey);
  const nsFromQuery = searchParams?.get("ns") || fullKey.split(":")[0] || "common";

  const [languages, setLanguages] = useState<LanguageItem[]>([]);
  const [values, setValues] = useState<ValueMap>({});
  const [versions, setVersions] = useState<VersionMap>({});
  const [loading, setLoading] = useState(true);
  const [saving, setSaving] = useState(false);
  const [toast, setToast] = useState<{ open: boolean; msg: string; sev: "success" | "error" | "info" | "warning"; }>({ open: false, msg: "", sev: "info" });

  // Yeni Anahtar Ekle dialog state
  const [createOpen, setCreateOpen] = useState(false);
  const [createNs, setCreateNs] = useState(nsFromQuery);
  const [createKey, setCreateKey] = useState("");
  const [createValues, setCreateValues] = useState<ValueMap>({});

  const mountedRef = useRef(false);
  useEffect(() => {
    mountedRef.current = true;
    return () => { mountedRef.current = false; };
  }, []);

  // Dilleri getir
  useEffect(() => {
    (async () => {
      try {
        
        const url = `${API_BASE}/api/v${API_VERSION}/culture/list`;
        const response = await fetchJson<any>(url);
        let languagesData: LanguageItem[] = [];
        if (Array.isArray(response)) languagesData = response;
        else if (response?.data && Array.isArray(response.data)) languagesData = response.data;
        else if (response?.items && Array.isArray(response.items)) languagesData = response.items;
        else if (response?.success && Array.isArray(response.data)) languagesData = response.data;

        if (!mountedRef.current) return;
        setLanguages(languagesData);
      } catch (err) {
        if (!mountedRef.current) return;
        setToast({ open: true, msg: t("detailPage.toast.languagesLoadError") || "Dil listesi alınamadı.", sev: "error" });
      }
    })();
  }, [t]);

  // Key değerlerini tüm diller için getir
  useEffect(() => {
    if (!languages.length) return;

    (async () => {
      setLoading(true);

      try {
 
        const promises = languages.map(async (lang) => {
  const url = `${API_BASE}/api/v${API_VERSION}/${lang.cultureCode}/localization/manager/search?ns=${encodeURIComponent(nsFromQuery)}&contains=${encodeURIComponent(fullKey)}`;

  const list = await fetchJson<TranslationDto[]>(url, {
    headers: {
      "Accept-Language": lang.cultureCode,
    },
  });

  const item = (Array.isArray(list) ? list : []).find((x) => x.key === fullKey);

  return {
    culture: lang.cultureCode,
    value: item?.value ?? "",
    version: item?.version ?? null,
  };
});

        const result = await Promise.all(promises);
        if (!mountedRef.current) return;

        const v: ValueMap = {};
        const ver: VersionMap = {};
        for (const r of result) {
          v[r.culture] = r.value ?? "";
          ver[r.culture] = r.version ?? null;
        }
        setValues(v);
        setVersions(ver);
      } catch (e) {
        if (!mountedRef.current) return;
        setToast({ open: true, msg: prettyApiError(e, t), sev: "error" });
      } finally {
        if (mountedRef.current) setLoading(false);
      }
    })();
  }, [languages, fullKey, nsFromQuery, t]);

  const handleValueChange = useCallback((culture: string, val: string) => {
    setValues((prev) => ({ ...prev, [culture]: val }));
  }, []);

  // Kaydet (Tüm Diller)
  const handleSaveAll = useCallback(async () => {
    setSaving(true);
    try {
      for (const lang of languages) {
        const culture = lang.cultureCode;
        const value = values[culture] ?? "";
        const expectedRowVersion = versions[culture] ?? null;

        const body = {
          key: fullKey,
          culture,
          value,
          expectedRowVersion,
          audit: { modifiedBy: "admin", reason: "Detail save", sourceIp: "127.0.0.1" },
        };

        const url = `${API_BASE}/api/v${API_VERSION}/${culture}/localization/manager/upsert`;
        try {
          await fetchJson(url, {
            method: "POST",
            body: JSON.stringify(body),
            headers: { "x-cache-invalidate": nsFromQuery, "x-audit-sensitive": "none" },
          });
        } catch (e: any) {
          setToast({ open: true, msg: `(${culture}) ${prettyApiError(e, t)}`, sev: "error" });
        }
      }

      // Kaydet sonrası taze versiyonları çek
      try {

          const promises = languages.map(async (lang) => {
  const url = `${API_BASE}/api/v${API_VERSION}/${lang.cultureCode}/localization/manager/search?ns=${encodeURIComponent(nsFromQuery)}&contains=${encodeURIComponent(fullKey)}`;
  const list = await fetchJson<TranslationDto[]>(url, {
    headers: {
      "Accept-Language": lang.cultureCode,
    },
  });
  const item = (Array.isArray(list) ? list : []).find((x) => x.key === fullKey);
  return { culture: lang.cultureCode, version: item?.version ?? null };
});

        const vers = await Promise.all(promises);
        const newVersions: VersionMap = {};
        for (const v of vers) newVersions[v.culture] = v.version ?? null;
        setVersions(newVersions);
      } catch { /* ignore */ }

      setToast({ open: true, msg: t("detailPage.toast.savedAllLanguages") || "Tüm diller için kaydedildi.", sev: "success" });
    } finally {
      setSaving(false);
    }
  }, [languages, values, versions, fullKey, nsFromQuery, t]);

  // Yeni Anahtar Ekle (Dialog → çoklu upsert)
  const handleCreateNewKey = useCallback(async () => {
    const ns = (createNs || "").trim();
    const k = (createKey || "").trim();
    if (!ns || !k) {
      setToast({ open: true, msg: t("detailPage.toast.namespaceKeyRequired") || "Namespace ve key zorunludur.", sev: "warning" });
      return;
    }
    const composedKey = `${ns}:${k}`;

    setSaving(true);
    try {
      for (const lang of languages) {
        const culture = lang.cultureCode;
        const value = createValues[culture] ?? "";
        const url = `${API_BASE}/api/v${API_VERSION}/${culture}/localization/manager/upsert`;
        const body = {
          key: composedKey,
          culture,
          value,
          expectedRowVersion: null,
          audit: { modifiedBy: "admin", reason: "Create new key", sourceIp: "127.0.0.1" },
        };
        try {
          await fetchJson(url, {
            method: "POST",
            body: JSON.stringify(body),
            headers: { "x-cache-invalidate": ns, "x-audit-sensitive": "none" },
          });
        } catch (e: any) {
          setToast({ open: true, msg: `(${culture}) ${prettyApiError(e, t)}`, sev: "error" });
        }
      }
      setToast({ open: true, msg: t("detailPage.toast.newKeyCreated") || "Yeni anahtar oluşturuldu.", sev: "success" });
      setCreateOpen(false);
      setCreateKey("");
      setCreateValues({});
    } finally {
      setSaving(false);
    }
  }, [languages, createNs, createKey, createValues, t]);

  const cultureOptions = useMemo(() => Array.from(new Set(languages.map((l) => l.cultureCode))), [languages]);

  return (
    <Box>
      <Breadcrumb
        title={t("detailPage.breadcrumb.title") || "Çeviri Detay"}
        subtitle={t("detailPage.breadcrumb.subtitle") || "Anahtar detayları ve tüm diller için değerler"}
        items={[
          { title: t("detailPage.breadcrumb.localization") || "Localization", to: `/${locale}/apps/localization` },
          { title: t("detailPage.breadcrumb.detail") || "Detail" },
        ]}
      />

      <Toolbar disableGutters sx={{ mb: 2, gap: 2, flexWrap: "wrap", alignItems: "center" }}>
        <Button variant="outlined" onClick={() => router.back()}>{t("detailPage.actions.backToList") || "← Listeye Dön"}</Button>
        <Box sx={{ flex: 1 }} />
        <Button variant="contained" onClick={() => setCreateOpen(true)}>{t("detailPage.actions.addNewKey") || "Yeni Anahtar Ekle"}</Button>
      </Toolbar>

      <Card sx={{ mb: 2 }}>
        <CardContent>
          <Typography variant="h6" gutterBottom>{t("detailPage.selectedKey.title") || "Seçili Anahtar"}</Typography>
          <Stack direction={{ xs: "column", sm: "row" }} spacing={2}>
            <TextField
              label={t("detailPage.fields.namespace") || "Namespace"}
              value={nsFromQuery}
              size="small"
              InputProps={{ readOnly: true }}
            />
            <TextField
              label={t("detailPage.fields.key") || "Key"}
              value={fullKey}
              size="small"
              InputProps={{ readOnly: true }}
              sx={{ flex: 1 }}
            />
          </Stack>
        </CardContent>
      </Card>

      <Card>
        <CardContent>
          <Stack direction="row" justifyContent="space-between" alignItems="center" sx={{ mb: 2 }}>
            <Typography variant="h6">{t("detailPage.allLanguages.title") || "Tüm Diller"}</Typography>
            <Button
              variant="contained"
              disabled={saving || loading}
              startIcon={saving ? <CircularProgress size={16} /> : undefined}
              onClick={handleSaveAll}
            >
              {t("detailPage.actions.saveAllLanguages") || "Kaydet (Tüm Diller)"}
            </Button>
          </Stack>

          {loading ? (
            <Box sx={{ display: "flex", justifyContent: "center", py: 6 }}>
              <CircularProgress />
            </Box>
          ) : (
            <Stack spacing={2} divider={<Divider />}>
              {languages.map((lang) => (
                <Stack key={lang.cultureCode} direction={{ xs: "column", sm: "row" }} spacing={2} alignItems="center">
                  <Box sx={{ width: 220, display: "flex", alignItems: "center" }}>
                    <span style={{ fontSize: 20, marginRight: 8 }}>{lang.flagEmoji ?? "🏳️"}</span>
                    <div>
                      <Typography variant="subtitle2">{lang.name}</Typography>
                      <Typography variant="caption" color="text.secondary">
                        {lang.cultureCode}
                        {lang.isDefault ? ` • ${t("detailPage.defaultLanguageBadge") || "Varsayılan"}` : ""}
                      </Typography>
                    </div>
                  </Box>
                  <TextField
                    fullWidth
                    size="small"
                    label={t("detailPage.fields.value") || "Value"}
                    value={values[lang.cultureCode] ?? ""}
                    onChange={(e) => handleValueChange(lang.cultureCode, e.target.value)}
                  />
                </Stack>
              ))}
            </Stack>
          )}
        </CardContent>
      </Card>

      {/* Yeni Anahtar Ekle Dialog */}
      <Dialog open={createOpen} onClose={() => setCreateOpen(false)} maxWidth="md" fullWidth>
        <DialogTitle>{t("newKeyDialog.title") || "Yeni Anahtar Ekle"}</DialogTitle>
        <DialogContent dividers>
          <Stack spacing={2} sx={{ mt: 1 }}>
            <Stack direction={{ xs: "column", sm: "row" }} spacing={2}>
              <TextField
                label={t("newKeyDialog.namespaceLabel") || "Namespace"}
                value={createNs}
                onChange={(e) => setCreateNs(e.target.value)}
                size="small"
                sx={{ minWidth: 200 }}
              />
              <TextField
                label={t("newKeyDialog.keyLabel") || "Key (ns olmadan)"}
                value={createKey}
                onChange={(e) => setCreateKey(e.target.value)}
                size="small"
                sx={{ flex: 1 }}
              />
            </Stack>

            <Typography variant="caption" color="text.secondary">
              {t("newKeyDialog.composedKeyLabel") || "Oluşacak tam anahtar:"} <b>{(createNs || "").trim()}:{(createKey || "").trim()}</b>
            </Typography>

            <Divider />

            <Typography variant="subtitle1">{t("newKeyDialog.valuesTitle") || "Değerler (Tüm Diller)"}</Typography>
            <Stack spacing={2} divider={<Divider />}>
              {languages.map((lang) => (
                <Stack key={lang.cultureCode} direction={{ xs: "column", sm: "row" }} spacing={2} alignItems="center">
                  <Box sx={{ width: 220, display: "flex", alignItems: "center" }}>
                    <span style={{ fontSize: 20, marginRight: 8 }}>{lang.flagEmoji ?? "🏳️"}</span>
                    <div>
                      <Typography variant="subtitle2">{lang.name}</Typography>
                      <Typography variant="caption" color="text.secondary">
                        {lang.cultureCode}
                        {lang.isDefault ? ` • ${t("newKeyDialog.defaultLanguageBadge") || "Varsayılan"}` : ""}
                      </Typography>
                    </div>
                  </Box>
                  <TextField
                    fullWidth
                    size="small"
                    label={t("newKeyDialog.valueLabel") || "Value"}
                    value={createValues[lang.cultureCode] ?? ""}
                    onChange={(e) =>
                      setCreateValues((prev) => ({ ...prev, [lang.cultureCode]: e.target.value }))
                    }
                  />
                </Stack>
              ))}
            </Stack>
          </Stack>
        </DialogContent>
        <DialogActions>
          <Button onClick={() => setCreateOpen(false)}>{t("common.cancel") || "İptal"}</Button>
          <Button variant="contained" onClick={handleCreateNewKey} disabled={saving}>
            {saving ? (t("common.saving") || "Kaydediliyor...") : (t("newKeyDialog.createButton") || "Oluştur")}
          </Button>
        </DialogActions>
      </Dialog>

      {/* Toast */}
      <Snackbar
        open={toast.open}
        autoHideDuration={3500}
        onClose={() => setToast((tState) => ({ ...tState, open: false }))}
        anchorOrigin={{ vertical: "bottom", horizontal: "right" }}
      >
        <Alert
          onClose={() => setToast((tState) => ({ ...tState, open: false }))}
          severity={toast.sev}
          variant="filled"
          sx={{ width: "100%" }}
        >
          {toast.msg}
        </Alert>
      </Snackbar>
    </Box>
  );
}
